⚡ 1. Entry Point: App.jsx

The root component that orchestrates global state providers and routing for the entire application.

App.jsx 
function App() {
  return (
    <ThemeProvider>           {/* 1. Dark/Light mode context */}
      <QueryClientProvider>  {/* 2. React Query (data fetching) */}
        <AuthProvider>       {/* 3. Auth context (logged-in user) */}
          <Router>           {/* 4. Browser routing */}
            <Toaster />      {/* 5. Global toast notifications */}
            <Navbar />       {/* 6. Persistent top navbar */}
            <Routes>
              {/* Public routes */}
              <Route path="/" element={<Home />} />
              <Route path="/items" element={<ItemList />} />

              {/* Protected routes (login required) */}
              <Route path="/dashboard" element={
                <ProtectedRoute><Dashboard /></ProtectedRoute>
              } />

              {/* Role-restricted routes */}
              <Route path="/verify-claims" element={
                <ProtectedRoute requiredRole={['staff', 'admin']}>
                  <VerifyClaims />
                </ProtectedRoute>
              } />
              <Route path="/admin" element={
                <ProtectedRoute requiredRole="admin">
                  <AdminDashboard />
                </ProtectedRoute>
              } />
            </Routes>
          </Router>
        </AuthProvider>
      </QueryClientProvider>
    </ThemeProvider>
  );
}

🔑 Key Mechanisms

  • Maintenance Heartbeat: Pings /api/health on mount to detect server status.
  • Provider Nesting: ThemeProvider → QueryClientProvider → AuthProvider ensures global state is available everywhere.
  • Route Protection: ProtectedRoute redirects unauthenticated users to /login.

🧠 2. Context Providers

AuthContext.jsx

Manages the global logged-in user state, persisting sessions via localStorage.

context/AuthContext.jsx 
export const AuthProvider = ({ children }) => {
    const [user, setUser] = useState(null);
    const [loading, setLoading] = useState(true);

    useEffect(() => {
        if (isAuthenticated()) {
            setUser(getCurrentUser()); // Restore from localStorage
        }
        setLoading(false);
    }, []);

    const login = (userData) => setUser(userData);
    const logout = () => {
        setUser(null);
        authLogout(); // Clears storage + redirects
    };
    const updateUser = (userData) => {
        setUser(userData);
        localStorage.setItem('user', JSON.stringify(userData));
    };

    return <AuthContext.Provider value={{ user, login, logout, updateUser, loading }}>
        {children}
    </AuthContext.Provider>;
};

ThemeContext.jsx

Handles dark/light mode switching with system preference detection.

context/ThemeContext.jsx 
export const ThemeProvider = ({ children }) => {
    const [theme, setTheme] = useState(() => {
        const saved = localStorage.getItem('theme');
        if (saved) return saved;
        return window.matchMedia('(prefers-color-scheme: dark)').matches
          ? 'dark' : 'light';
    });

    useEffect(() => {
        document.documentElement.classList.toggle('dark', theme === 'dark');
        localStorage.setItem('theme', theme);
    }, [theme]);

    const toggleTheme = () => setTheme(p => p === 'light' ? 'dark' : 'light');

    return <ThemeContext.Provider value={{ theme, toggleTheme }}>
        {children}
    </ThemeContext.Provider>;
};

🔌 3. API Layer: api.js

A single Axios instance used by every service — with automatic JWT attachment and global error handling.

services/api.js 
const api = axios.create({ baseURL: 'http://localhost:5000/api' });

// REQUEST: Auto-attach JWT token
api.interceptors.request.use((config) => {
    const token = localStorage.getItem('token');
    if (token) config.headers.Authorization = `Bearer ${token}`;
    return config;
});

// RESPONSE: Handle global errors
api.interceptors.response.use(
    (response) => response,
    (error) => {
        if (error.response?.status === 401) {
            localStorage.removeItem('token');
            window.location.href = '/login'; // Expired session
        }
        if (error.response?.status === 503 && error.response.data.isMaintenance) {
            if (user.role !== 'admin') window.location.href = '/maintenance';
        }
        return Promise.reject(error);
    }
);

✅ Why This Matters

  • You never manually add Authorization headers in components.
  • Expired sessions are automatically handled app-wide.

🧩 5. Reusable UI Components

ProtectedRoute.jsx

A route guard wrapping pages that need login or a specific role.

const ProtectedRoute = ({ children, requiredRole }) => {
    const { user, loading } = useAuth();
    if (loading) return <LoadingSpinner />;
    if (!user) return <Navigate to="/login" />;
    if (requiredRole) {
        const allowed = Array.isArray(requiredRole)
          ? requiredRole : [requiredRole];
        if (user.role !== 'admin' && !allowed.includes(user.role)) {
            return <Navigate to="/dashboard" />;
        }
    }
    return children;
};

Navbar.jsx — Persistent Navigation with Live Notifications

The top navigation bar is always visible. It adapts its links based on the user's role and polls for unread notifications every 30 seconds.

components/Navbar.jsx 
const Navbar = () => {
    const { user, logout } = useAuth();
    const { pathname } = useLocation();       // current URL path
    const [unreadCount, setUnreadCount] = useState(0);

    // Poll for unread notifications every 30 seconds
    useEffect(() => {
        if (!user) return;
        const fetchCount = async () => {
            const data = await getUnreadCount();
            setUnreadCount(data.count);
        };
        fetchCount();
        const interval = setInterval(fetchCount, 30000); // re-check every 30s
        return () => clearInterval(interval);             // cleanup on unmount
    }, [user]);

    // Highlight the active page link
    const isActive = (path) => pathname === path;

    return (
        <nav className="sticky top-0 z-50">
            <Link to="/">Campus<span>Find</span></Link>

            {user ? (
                <>
                    {/* Role-conditional navigation links */}
                    <Link to="/dashboard"
                          className={isActive('/dashboard') ? 'active' : ''}>
                        Dashboard
                    </Link>
                    {(user.role === 'staff' || user.role === 'admin') && (
                        <Link to="/verify-claims">Verify Claims</Link>
                    )}
                    {user.role === 'admin' && (
                        <Link to="/admin">Admin</Link>
                    )}

                    {/* Notification bell with live unread count */}
                    <Link to="/notifications">
                        <Bell />
                        {unreadCount > 0 && <span>{unreadCount}</span>}
                    </Link>

                    <button onClick={logout}>Logout</button>
                </>
            ) : (
                <>
                    <Link to="/login">Login</Link>
                    <Link to="/register">Register</Link>
                </>
            )}
        </nav>
    );
};

🔍 How It Works

  • Role-aware links: Staff see "Verify Claims". Admins see both "Verify Claims" and "Admin".
  • Active highlighting: isActive() compares the current URL path to style the active link.
  • Live badge: setInterval polls getUnreadCount() every 30s so the bell icon shows a live count.
  • Sticky positioning: The navbar stays at the top as the user scrolls.

📄 6. Key Pages

Dashboard.jsx — Personal Hub

The main landing page after login. Fetches the user's items, claims, and notifications in parallel for speed, then computes statistics from the results.

pages/Dashboard.jsx 
const Dashboard = () => {
    const { user } = useAuth();
    const [myItems, setMyItems] = useState([]);
    const [myClaims, setMyClaims] = useState([]);
    const [notifications, setNotifications] = useState([]);

    useEffect(() => {
        // Fetch all 3 data sources IN PARALLEL for speed
        const [items, claims, notifs] = await Promise.all([
            getMyItems(),
            getMyClaims(),
            getNotifications(5),       // latest 5 only
        ]);
        setMyItems(items.items || []);
        setMyClaims(claims);
        setNotifications(notifs);
    }, []);

    // Computed statistics from fetched data
    const matchedItemsCount = myItems.filter(
        i => i.matchedItems?.length > 0
    ).length;
    const unreadCount = notifications.filter(n => !n.read).length;

    return (
      <>
        {/* Welcome banner with user's name */}
        <h1>Welcome back, {user?.name}! 👋</h1>

        {/* Quick Action Cards — navigate on click */}
        <Link to="/report-item">
            <ActionCard title="Report an Item" icon={<Plus />} />
        </Link>
        <Link to="/items">
            <ActionCard title="Browse Items" icon={<Search />} />
        </Link>

        {/* Statistics Row — 4 stat cards */}
        <StatCard label="My Items" value={myItems.length} />
        <StatCard label="Matches Found" value={matchedItemsCount} />
        <StatCard label="My Claims" value={myClaims.length} />
        <StatCard label="Unread Alerts" value={unreadCount} />

        {/* Recent Items Grid — shows top 3 */}
        {myItems.slice(0, 3).map(item =>
            <ItemCard key={item._id} item={item} />
        )}

        {/* Recent Notifications List */}
        {notifications.map(n =>
            <NotificationRow key={n._id} notif={n} />
        )}
      </>
    );
};

⚡ Performance Pattern

  • Promise.all: All 3 API calls run simultaneously instead of sequentially — this makes the Dashboard load ~3x faster.
  • Computed stats: matchedItemsCount and unreadCount are derived from the fetched data, not separate API calls.
  • Slice(0, 3): Only the 3 most recent items are displayed in the grid preview.

📸 How Image Upload Works

  • The user selects images via a file input. JavaScript creates a FormData object.
  • Text fields are appended as key-value pairs. Each image is appended as 'images'.
  • The request is sent with Content-Type: multipart/form-data so Express + Multer can process the files on the backend.
  • Images are saved to the /uploads directory and their paths are stored in the Item document.

VerifyClaims.jsx — Staff Claim Review

Staff-only page to review, approve, or reject pending claims. Includes status filter tabs and proof viewing.

pages/VerifyClaims.jsx 
const VerifyClaims = () => {
    const [activeTab, setActiveTab] = useState('pending');
    const [claims, setClaims] = useState([]);

    useEffect(() => {
        // Fetch claims filtered by the active tab
        const data = await getClaims({ status: activeTab });
        setClaims(data);
    }, [activeTab]);

    return (
      <>
        {/* Filter tabs: All | Pending | Approved | Rejected */}
        <div className="tabs">
            {['all', 'pending', 'approved', 'rejected'].map(tab =>
                <button key={tab}
                    className={activeTab === tab ? 'active-tab' : ''}
                    onClick={() => setActiveTab(tab)}>
                    {tab}
                </button>
            )}
        </div>

        {/* Claim Cards — each shows item, proof, and actions */}
        {claims.map(claim => (
            <div key={claim._id} className="claim-card">
                <ItemCard item={claim.item} plain />
                <p>Proof: {claim.proofDescription}</p>
                {claim.proofImages?.map(img =>
                    <img src={getImageUrl(img)} />
                )}

                {/* Staff action buttons */}
                {claim.status === 'pending' && (
                    <>
                        <button onClick={() => updateClaimStatus(
                            claim._id, { status: 'approved' }
                        )}>
                            ✅ Approve
                        </button>
                        <button onClick={() => updateClaimStatus(
                            claim._id, {
                                status: 'rejected',
                                verificationNotes: 'Proof insufficient'
                            }
                        )}>
                            ❌ Reject
                        </button>
                    </>
                )}
            </div>
        ))}
      </>
    );
};

🛡️ Approval Flow

  • When approved: item status changes to 'claimed', and the item reporter gets a notification.
  • When rejected: staff must provide a verificationNotes reason explaining why.
  • The verifiedBy field automatically records which staff member made the decision.

});

app.use(maintenanceMode);                       // 7. Block non-admins
app.use('/api/auth', authRoutes);               // 8. Route groups
app.use('/api/items', itemRoutes);
app.use('/api/claims', claimRoutes);
app.use('/api/notifications', notificationRoutes);
app.use('/api/admin', adminRoutes);
app.use(errorHandler);                          // 9. Central error handler

🔄 9. Data Lifecycle Flow

Scenario: A student loses a backpack and eventually gets it back.

Step 1 — Student Logs In

POST /api/auth/login → JWT stored in localStorage

Step 2 — Reports Lost Item

POST /api/items with FormData (image attached) → Item created: type='lost', status='active'

Step 3 — Matching Runs

Backend finds a 'found' item with similar properties → links via matchedItems[] → Notification sent

Step 4 — Finder Submits Claim

POST /api/claims with proof description & images → Claim: status='pending'

Step 5 — Staff Reviews

GET /api/claims?status=pending → Staff sees proof details

Step 6 — Claim Approved

PATCH /api/claims/:id → Item status → 'claimed' → Student notified

Step 7 — Item Returned ✅

Student views notification in-app. The item is returned to its owner.
